createScene(1000, 700);

for(int::y=0; y<5*8; y=y+5) then
    for(int::x=0; x<5*8; x=x+5) then
        double::xd = (x+y)+0.0;

        if( (x+y)/2 == xd/2 ) then
            createBox(x,y,0, 4,4,4, 1,1,0);
        end else then
            createBox(x,y,0, 4,4,4, 1,0,0);
        end
    end
end

buildScene();
